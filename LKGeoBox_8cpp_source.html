95TVector3 LKGeoBox::GetCorner(Int_t idx) const
96{
97 // 0 1 2 3 4 5 6 7

178bool LKGeoBox::TestPointInsidePlane(int iPlane, TVector3 point) const
179{
180 Int_t idx = iPlane;
181 // idx : 0 1 2 3 4 5
182 // face: +x -x +y -y +z -z
183 Int_t cidx1[] = {0, 4, 0, 2, 3, 1};
184 Int_t cidx2[] = {2, 6, 6, 4, 7, 5};
185 Int_t cidx3[] = {1, 5, 7, 3, 4, 2};
186 bool condition1, condition2;
187
188 TVector3 a(GetCorner(cidx3[idx]));
189 TVector3 b(GetCorner(cidx1[idx]));
190 TVector3 d(GetCorner(cidx2[idx]));
191
192 TVector3 am = point - a;
193 TVector3 ab = b - a;
194 condition1 = (0<(am*ab));
195 condition2 = ((am*ab)<(ab*ab));
196 if ((condition1 && condition2)==false) {
197 return false;
198 }
199 TVector3 ad = d - a;
200 condition1 = (0<(am*ad));
201 condition2 = ((am*ad)<(ad*ad));
202 if ((condition1 && condition2)==false) {
203 return false;
204 }
205 return true;
206}
