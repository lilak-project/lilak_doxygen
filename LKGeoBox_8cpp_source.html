85TVector3 LKGeoBox::GetCorner(Int_t idx) const
86{
87 // 0 1 2 3 4 5 6 7
88 // +x +x +x +x -x -x -x -x
89 // +y +y -y -y -y -y +y +y
90 // +z -z -z +z +z -z -z +z
91 Int_t xpm[] = {1, 1, 1, 1, -1,-1,-1,-1};
92 Int_t ypm[] = {1, 1,-1,-1, -1,-1, 1, 1};
93 Int_t zpm[] = {1,-1,-1, 1, 1,-1,-1, 1};
94
95 return TVector3(.5*xpm[idx]*fdX+fX, .5*ypm[idx]*fdY+fY, .5*zpm[idx]*fdZ+fZ);
96}
