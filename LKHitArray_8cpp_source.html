610LKGeoHelix LKHitArray::FitHelix(axis_t ref)
611{
612 auto circle = FitCircle(ref);
613
614 LKGeoHelix helix;
615 helix.SetRMS (-1);
616 helix.SetRMSR(-1);
617 helix.SetRMST(-1);
618
619 if (fN < 5)
620 return helix;
621
622 SortByX(true);
623 SortByY(true);
624 SortByZ(true);
625
626 auto helixI = circle.GetX();
627 auto helixJ = circle.GetY();
628
629 auto pos0 = GetHit(0) -> GetPosition(ref);
630 auto i = pos0.I() - helixI;
631 auto j = pos0.J() - helixJ;
632 auto axisI = TVector2( i,j).Unit();
633 auto axisJ = TVector2(-j,i).Unit();
634
635 auto meanK = 0.; // <k>
636 auto meanAlpha = 0.; // <alpha>
637 auto meanAlpha2 = 0.; // <alpha*alpha>
638 auto meanAlphaK = 0.; // <alpha*k>
639
640 auto alphaStack = 0.; // alpha angle stack (save point)
641 auto alphaInit = TMath::ATan2(j, i); // alpha angle of first hit
642 auto alphaMin = alphaInit; // maximum value of alpha angle
643 auto alphaMax = alphaInit; // minimum value of alpha angle
644
645 auto halfOfPi = .5*TMath::Pi();
646
647 TIter nextHit(this);
648 LKHit *hit;
649 while ((hit = (LKHit *) nextHit()))
650 {
651 auto w = hit -> GetCharge();
652 auto p = hit -> GetPosition(ref);
653 auto v = TVector2(p.I()-helixI,p.J()-helixJ);
654 auto rotI = v*axisI;
655 auto rotJ = v*axisJ;
656
657 auto alpha = TMath::ATan2(rotJ, rotI);
658 if (alpha > halfOfPi || alpha < -halfOfPi) { // if the angle is over +-pi/2 relative to alphaStack
659 alpha += alphaStack;
660 alphaStack = alpha;
661 axisI = v.Unit();
662 axisJ = TVector2(-v.Y(), v.X()).Unit();
663 } else
664 alpha += alphaStack;
665
666 alpha += alphaInit;
667 if (alpha < alphaMin) alphaMin = alpha;
668 if (alpha > alphaMax) alphaMax = alpha;
669 hit -> SetAlpha(alpha);
670
671 meanK += w * p.K();
672 meanAlpha += w * alpha;
673 meanAlpha2 += w * alpha * alpha;
674 meanAlphaK += w * alpha * p.K();
675 }
676
677 meanK /= fW;
678 meanAlpha /= fW;
679 meanAlpha2 /= fW;
680 meanAlphaK /= fW;
681
682 Double_t slope = (meanAlphaK - meanAlpha*meanK) / (meanAlpha2 - meanAlpha*meanAlpha);
683 Double_t offset = (meanAlpha2*meanK - meanAlpha*meanAlphaK) / (meanAlpha2 - meanAlpha*meanAlpha);
684
685 if (std::isinf(slope))
686 return helix;
687
688 helix.SetI(helixI);
689 helix.SetJ(helixJ);
690 helix.SetR(circle.GetR());
691 helix.SetH(alphaMax);
692 helix.SetT(alphaMin);
693 helix.SetS(slope);
694 helix.SetK(offset);
695 helix.SetA(ref);
696
697 Double_t sr = 0; // = sum of distance from hit to track in radial axis
698 Double_t st = 0; // = sum of distance from hit to track in axial axis
699 Double_t ss = 0; // = sum of distance from hit to track
700
701 nextHit.Begin();
702 while ((hit = (LKHit *) nextHit()))
703 {
704 auto w = hit -> GetCharge();
705 TVector3 mp = helix.HelicoidMap(hit->GetPosition(),hit->GetAlpha());
706 auto xx = mp.X() * mp.X();
707 auto yy = mp.Y() * mp.Y();
708
709 sr += w * xx;
710 st += w * yy;
711 ss += w * (xx + yy);
712 }
713
714 helix.SetRMS (sqrt(ss/(fW*(1-3/fN))));
715 helix.SetRMSR(sqrt(sr/(fW*(1-3/fN))));
716 helix.SetRMST(sqrt(st/(fW*(1-3/fN))));
717
718 return helix;
719}

790bool LKHitArray::Subtract(Double_t x, Double_t y, Double_t z, Double_t w)
791{
792 Double_t wsum = fW - w;
793
794 fEX = (fW * fEX - w * x) / wsum;
795 fEY = (fW * fEY - w * y) / wsum;
796 fEZ = (fW * fEZ - w * z) / wsum;
797 fEXX = (fW * fEXX - w * x * x) / wsum;
798 fEYY = (fW * fEYY - w * y * y) / wsum;
799 fEZZ = (fW * fEZZ - w * z * z) / wsum;
800 fEXY = (fW * fEXY - w * x * y) / wsum;
801 fEYZ = (fW * fEYZ - w * y * z) / wsum;
802 fEZX = (fW * fEZX - w * z * x) / wsum;
803
804 fW = wsum;
805 --fN;
806
807 return true;
808}
