31class LKTbIterationParameters
32{
33 public:
34 LKTbIterationParameters() {}
35
36 double fC1 = DBL_MAX;
37 double fC2 = DBL_MAX;
38 double fT1 = 0;
39 double fT2 = 0;
40 double fScaleTbStep0 = 1;
41 double fScaleTbStep = 1;
42
43 void SetScaleTbStep(double value) {
44 fScaleTbStep0 = value;
45 fScaleTbStep = value;
46 }
47
48 double Slope() const {
49 double slope = - (fC1 - fC2) / (fT1 - fT2);
50 return slope;
51 }
52
53 double TbStep() const {
54 double tbStep = fScaleTbStep * Slope();
55 return tbStep;
56 }
57
58 double NextTb(double tb) const {
59 double step = TbStep();
60 if (step> 1) step = 1;
61 else if (step<-1) step = -1;
62 double tbNext = tb + step;
63 return tbNext;
64 }
65
66 bool Add(double c0, double t0) {
67 if (c0<fC1) {
68 fC2 = fC1;
69 fT2 = fT1;
70 fC1 = c0;
71 fT1 = t0;
72 fScaleTbStep = fScaleTbStep0;
73 return true;
74 }
75 else if (c0<fC2) {
76 fC2 = c0;
77 fT2 = t0;
78 fScaleTbStep = fScaleTbStep0;
79 return true;
80 }
81 else {
82 fScaleTbStep = 0.5 * fScaleTbStep;
83 return false;
84 }
85 }
