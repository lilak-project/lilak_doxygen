210Int_t LHPadPlaneYCut::FindPadID(Double_t i, Double_t j)
211{
212 Int_t section = FindSection(i,j);
213
214 Double_t xRotatedToSec0 = i*fCosPiNo4[section] + j*fSinPiNo4[section];
215 Double_t yRotatedToSec0 = -i*fSinPiNo4[section] + j*fCosPiNo4[section];
216 Double_t rRotatedToSec0 = sqrt(xRotatedToSec0*xRotatedToSec0 + yRotatedToSec0*yRotatedToSec0);
217
218 Double_t rFromSectionBottom = rRotatedToSec0 - fRadiusLayer0 + .5*fPadHeight;
219 if (rFromSectionBottom < 0)
220 return -1;
221
222 Int_t layer = (Int_t)(rFromSectionBottom/fRSpacing);
223 if (layer > fNumLayers)
224 return -1;
225
226 Int_t pm = 1;
227 if (xRotatedToSec0 < 0) {
228 xRotatedToSec0 = -xRotatedToSec0;
229 pm = -1;
230 }
231
232 if (xRotatedToSec0 < .5*fPadGap)
233 return -1;
234
235 Double_t xFromRow0LeftEdge = xRotatedToSec0 - .5*fPadGap;
236 Int_t row = (Int_t) (xFromRow0LeftEdge / fXSpacing) + 1;
237 if (xFromRow0LeftEdge - (row-1)*fXSpacing > fPadWidth)
238 return -1;
239
240 row = row - fNumSkippedHalfRows[layer];
241
242 if (row > fNumHalfRowsInLayer[layer])
243 return -1;
244
245 return FindPadID(section,pm*row,layer);
246}
