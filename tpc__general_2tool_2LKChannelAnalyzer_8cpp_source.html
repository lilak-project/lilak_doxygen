348void LKChannelAnalyzer::FitAmplitude(double *buffer, double tbStartOfPulse,
349 int &ndf,
350 double &amplitude,
351 double &chi2NDF)
352{
353 double refy = 0;
354 double ref2 = 0;
355 int tb0 = int(tbStartOfPulse);
356 double tbOffset = tbStartOfPulse - tb0;
357
358 int ndfFit = 0;
359 int iTbPulse = 0;
360 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
361 {
362 int tbData = tb0 + iTbPulse;
363 //if (tbData<0 || tbData>=350)
364 // lk_debug << tbData << endl;
365 if (tbData>=fTbMax)
366 break;
367 double valueData = buffer[tbData];
368 if (valueData>=fDynamicRange)
369 continue;
370 ndfFit++;
371 double tbRef = iTbPulse - tbOffset + 0.5;
372 double valueRef = fPulse -> Eval(tbRef);
373 double errorRef = fPulse -> Error(tbRef);
374 double weigth = 1./(errorRef*errorRef);
375 refy += weigth * valueRef * valueData;
376 ref2 += weigth * valueRef * valueRef;
377 }
378 ndfFit = ndfFit-2;
379#ifdef DEBUG_CHANA_FITAMPLITUDE
380 lk_debug << "ndf is " << ndf << " ndfFit is " << ndfFit << endl;
381#endif
382
383 if (ref2==0) {
384 chi2NDF = 1.e10;
385 return;
386 }
387 amplitude = refy / ref2;
388 chi2NDF = 0;
389
390 ndfFit = 0;
391 iTbPulse = 0;
392 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
393 {
394 int tbData = tb0 + iTbPulse;
395 //if (tbData<0 || tbData>=350)
396 // lk_debug << "e " << tbData << endl;
397 if (tbData>=fTbMax)
398 break;
399 double valueData = buffer[tbData];
400 if (valueData>=fDynamicRange)
401 continue;
402 ndfFit++;
403 double tbRef = iTbPulse - tbOffset + 0.5;
404 double valueRefA = amplitude * fPulse -> Eval(tbRef);
405 double errorRefA = amplitude * fPulse -> Error(tbRef);
406 double residual = (valueData-valueRefA)*(valueData-valueRefA)/errorRefA/errorRefA;
407 chi2NDF += residual;
408 }
409 ndfFit = ndfFit-2;
410
411 //lk_debug << "ndfFit: " << ndfFit << endl;
412
413 chi2NDF = chi2NDF/ndfFit;
414 ndf = ndfFit;
415}
