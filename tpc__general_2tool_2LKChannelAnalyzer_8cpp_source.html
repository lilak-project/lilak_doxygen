224double LKChannelAnalyzer::FindAndSubtractPedestal(double *buffer)
225{

272 pedestalMeanRef = 0.5 * (fPedestalSample[iSample] + fPedestalSample[jSample]);
273 idx1 = iSample;
274 idx2 = jSample;
275 }
276 }
277 }
278 }
279 else {
280 for (auto iSample=0; iSample<fNumPedestalSamples; ++iSample) {
281 for (auto jSample=0; jSample<fNumPedestalSamples; ++jSample) {
282 if (iSample>=jSample) continue;
283 double diff = abs(fPedestalSample[iSample] - fPedestalSample[jSample]);
284 if (diff<pedestalDiffMin) {
285 pedestalDiffMin = diff;
286 pedestalMeanRef = 0.5 * (fPedestalSample[iSample] + fPedestalSample[jSample]);
287 idx1 = iSample;
288 idx2 = jSample;

290 }
291 }
292 }
293
294 //double pedestalErrorRef = 0.1 * pedestalMeanRef;
295 //double pedestalErrorRefSample = 0.2 * pedestalMeanRef;
296 double pedestalErrorRefSample = 0.1 * pedestalMeanRef;

299 pedestalErrorRefSample = fPedestalErrorRefSampleCut;

301#ifdef DEBUG_CHANA_FINDPED

303 lk_debug << "ref-diff : " << pedestalMeanRef << endl;
304 //lk_debug << "ref-error: " << pedestalErrorRef << endl;
305 lk_debug << "ref-error-part: " << pedestalErrorRefSample << endl;
306#endif
307
308 double pedestalFinal = 0;
309 int countNumPedestalTb = 0;
310
311 tbGlobal = 0;
312 for (auto iSample=0; iSample<fNumPedestalSamplesM1; ++iSample)
313 {
314 double diffSample = abs(pedestalMeanRef - fPedestalSample[iSample]);
315 if (diffSample<pedestalErrorRefSample)
316 {
317 fUsedSample[iSample] = true;
318 countNumPedestalTb += fNumTbSample;
319 for (int iTb=0; iTb<fNumTbSample; iTb++)
320 {
321 pedestalFinal += buffer[tbGlobal];
322 tbGlobal++;
323 }
324#ifdef DEBUG_CHANA_FINDPED
325 lk_debug << iSample << " diff=" << diffSample << " " << pedestalFinal/countNumPedestalTb << " " << countNumPedestalTb << endl;
326#endif
327 }
328 else
329 tbGlobal += fNumTbSample;
330 }
331 double diffSample = abs(pedestalMeanRef - fPedestalSample[fNumPedestalSamplesM1]);
332 if (diffSample<pedestalErrorRefSample)
333 {
334 fUsedSample[fNumPedestalSamplesM1] = true;
335 countNumPedestalTb += fNumTbSampleLast;
336 for (int iTb=0; iTb<fNumTbSampleLast; iTb++) {
337 pedestalFinal += buffer[tbGlobal];
338 tbGlobal++;
339 }
340#ifdef DEBUG_CHANA_FINDPED
341 lk_debug << fNumPedestalSamplesM1 << " diff=" << diffSample << " " << pedestalFinal/countNumPedestalTb << " " << countNumPedestalTb << endl;
342#endif
343 }
344
345 pedestalFinal = pedestalFinal / countNumPedestalTb;
346
347#ifdef DEBUG_CHANA_FINDPED
348 lk_debug << pedestalFinal << endl;
349#endif
350 for (auto iTb=0; iTb<fTbMax; ++iTb)
351 buffer[iTb] = buffer[iTb] - pedestalFinal;
352
353 fDynamicRange = fDynamicRange - pedestalFinal;
354
355 fPedestal = pedestalFinal;
356
357 return pedestalFinal;

612void LKChannelAnalyzer::FitAmplitude(double *buffer, double tbStartOfPulse,
613 int &ndf,
614 double &amplitude,
615 double &chi2NDF)
616{
617 if (isnan(tbStartOfPulse)) {
618#ifdef DEBUG_CHANA_FITAMPLITUDE
619 lk_debug << "return tb is nan" << endl;
620#endif
621 chi2NDF = 1.e10;
622 return;
623 }
624#ifdef DEBUG_CHANA_FITAMPLITUDE
625 lk_debug << "pulse: "; for (auto i=0; i<20; ++i) e_cout << i << "/" << fPulse -> Eval(i) << ", "; e_cout << endl;
626#endif
627 double refy = 0;
628 double ref2 = 0;
629 int tb0 = int(tbStartOfPulse);
630#ifdef DEBUG_CHANA_FITAMPLITUDE
631 lk_debug << tb0 << " = abs(" << tbStartOfPulse << ")" << endl;
632#endif
633 double tbOffset = tbStartOfPulse - tb0;
634
635 int ndfFit = 0;
636 int iTbPulse = 0;
637 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
638 {
639 int tbData = tb0 + iTbPulse;
640#ifdef DEBUG_CHANA_FITAMPLITUDE
641 lk_debug << tbData << " = " << tb0 << " " << iTbPulse << endl;
642#endif
643 //if (tbData<0 || tbData>=350)
644 // lk_debug << tbData << endl;
645 if (tbData>=fTbMax)
646 break;
647 double valueData = buffer[tbData];
648 if (valueData>=fDynamicRange)
649 continue;
650 ndfFit++;
651 double tbRef = iTbPulse - tbOffset + 0.5;
652 double valueRef = fPulse -> Eval(tbRef);
653 double errorRef = fPulse -> Error(tbRef);
654 double weigth = 1./(errorRef*errorRef);
655 refy += weigth * valueRef * valueData;
656 ref2 += weigth * valueRef * valueRef;
657#ifdef DEBUG_CHANA_FITAMPLITUDE
658 lk_debug << iTbPulse << "; " << refy << " / " << ref2 << endl;
659#endif
660 }
661 ndfFit = ndfFit-2;
662#ifdef DEBUG_CHANA_FITAMPLITUDE
663 //lk_debug << "ndf is " << ndf << " ndfFit is " << ndfFit << endl;
664#endif
665
666 if (ref2==0) {
667 chi2NDF = 1.e10;
668 return;
669 }
670 amplitude = refy / ref2;
671#ifdef DEBUG_CHANA_FITAMPLITUDE
672 lk_debug << "amplitude = " << amplitude << " = " << refy << " / " << ref2 << endl;
673#endif
674 chi2NDF = 0;
675
676 ndfFit = 0;
677 iTbPulse = 0;
678 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
679 {
680 int tbData = tb0 + iTbPulse;
681 //if (tbData<0 || tbData>=350)
682 // lk_debug << "e " << tbData << endl;
683 if (tbData>=fTbMax)
684 break;
685 double valueData = buffer[tbData];
686 if (valueData>=fDynamicRange)
687 continue;
688 ndfFit++;
689 double tbRef = iTbPulse - tbOffset + 0.5;
690 double valueRefA = amplitude * fPulse -> Eval(tbRef);
691 double errorRefA = amplitude * fPulse -> Error(tbRef);
692 double residual = (valueData-valueRefA)*(valueData-valueRefA)/errorRefA/errorRefA;
693#ifdef DEBUG_CHANA_FITAMPLITUDE
694 lk_debug << residual << " = (" << valueData << " - " << valueRefA << ")^2 / " << errorRefA << "^2 at " << tbRef << endl;
695#endif
696 chi2NDF += residual;
697 }
698 ndfFit = ndfFit-2;
699
700 //lk_debug << "ndfFit: " << ndfFit << endl;
701
702 chi2NDF = chi2NDF/ndfFit;
703 ndf = ndfFit;
704}
