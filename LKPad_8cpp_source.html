131Int_t LKPad::Compare(const TObject *obj) const
132{
140
141 auto pad2 = (LKPad *) obj;
142
143 int sortThisPadLatterThanPad2 = 1;
144 int sortThisPadEarlierThanPad2 = -1;
145 int noChange = 0;
146
147 if (fSortValue >= 0) {
148 if (fSortValue < pad2 -> GetSortValue()) return sortThisPadEarlierThanPad2;
149 else if (fSortValue > pad2 -> GetSortValue()) return sortThisPadLatterThanPad2;
150 return noChange;
151 }
152 if (pad2 -> GetLayer() < fLayer) return sortThisPadEarlierThanPad2;
153 else if (pad2 -> GetLayer() > fLayer) return sortThisPadLatterThanPad2;
154 else //same layer
155 {
156 if (pad2 -> GetSection() > fSection) return sortThisPadEarlierThanPad2;
157 else if (pad2 -> GetSection() < fSection) return sortThisPadLatterThanPad2;
158 else // same layer, same section
159 {
160 if (pad2 -> GetRow() < fRow) return sortThisPadEarlierThanPad2;
161 else if (pad2 -> GetRow() > fRow) return sortThisPadLatterThanPad2;
162 else // same pad
163 return noChange;
164 }
165 }

275void LKPad::FillBufferIn(Int_t tbin, Double_t val, Int_t trackID)
276{
277 fActive = true;
278 fBufferIn[tbin] += val;
279 if (trackID != -1) {
280 Int_t numTracks = fMCIDArray.size();
281 Int_t idxFound = -1;
282 for (Int_t idx = 0; idx < numTracks; ++idx) {
283 if (fMCIDArray[idx] == trackID) {
284 idxFound = idx;
285 break;
286 }
287 }
288 if (idxFound == -1) {
289 fMCIDArray.push_back(trackID);
290 fMCWeightArray.push_back(val);
291 fMCTbArray.push_back(tbin);
292 }
293 else {
294 auto w = fMCWeightArray[idxFound];
295 auto tb = fMCTbArray[idxFound];
296
297 auto ww = w + val;
298 fMCWeightArray[idxFound] = ww;
299
300 tb = (w*tb + val*tbin)/ww;
301 fMCTbArray[idxFound] = tb;
302 }
303 }
304}
