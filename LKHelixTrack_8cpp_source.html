430void LKHelixTrack::DetermineParticleCharge(TVector3 vertex)
431{
432 Double_t lHead = ExtrapolateToAlpha(fH);
433 Double_t lTail = ExtrapolateToAlpha(fT);
434
435 TVector3 q;
436 Double_t alpha;
437 Double_t lVertex = ExtrapolateToPointAlpha(vertex, q, alpha);
438
439 if (std::abs(lVertex - lTail) > std::abs(lVertex - lHead)) {
440 auto a = fT;
441 fT = fH;
442 fH = a;
443 }
444
445 if (fT < fH) {
446 if (fS > 0) fIsPositiveChargeParticle = false;
447 else fIsPositiveChargeParticle = true;
448 }
449 else {
450 if (fS > 0) fIsPositiveChargeParticle = true;
451 else fIsPositiveChargeParticle = false;
452 }
453}

611{

613 return false;
614

616 Double_t j1 = fJ + jOff;
617 Double_t j2 = fJ - jOff;
618

622 Double_t d1Temp = d1Cand;

626 d1Temp = std::abs(alpha1Temp-fH);

628 break;
629 else {
630 alpha1 = alpha1Temp;
631 d1Cand = d1Temp;
632 }
633 }
634 while (1) {
635 alpha1Temp = alpha1 - 2*TMath::Pi();
636 d1Temp = std::abs(alpha1Temp-fH);

902Double_t
903LKHelixTrack::ExtrapolateByMap(TVector3 p, TVector3 &q, TVector3 &m) const
904{
905 Double_t lHead = ExtrapolateToAlpha(fH);
906 Double_t lTail = ExtrapolateToAlpha(fT);
907 Double_t lOff = lHead;
908 if (lHead > lTail)
909 lOff = lTail;
910
911 Double_t alpha;
912 Double_t length = ExtrapolateToPointAlpha(p, q, alpha);
913
914 LKVector3 p2(p, fA);
915 LKVector3 q2(q, fA);
916
917 Double_t di = p2.I() - fI;
918 Double_t dj = p2.J() - fJ;
919 Double_t dr = sqrt(di*di + dj*dj) - fR;
920 Double_t dk = p2.K() - q2.K();
921
922 m = TVector3(dr, dk/CosDip(), length+dk*TMath::Sin(DipAngle()) - lOff);
923
924 return alpha * fR / CosDip();
925}

927 Double_t
928LKHelixTrack::Continuity(Double_t &totalLength, Double_t &continuousLength)
929{
930 Int_t numHits = fHitArray.GetNumHits();
931 if (numHits < 2)
932 return -1;
933
934 SortHits();
935
936 Double_t total = 0;
937 Double_t continuous = 0;
938 TVector3 before = Map(fHitArray.GetHit(0)->GetPosition());
939
940 for (auto iHit = 1; iHit < numHits; iHit++)
941 {
942 TVector3 current = Map(fHitArray.GetHit(iHit)->GetPosition());
943 auto length = std::abs(current.Z()-before.Z());
944
945 total += length;
946 if (length < 25)
947 continuous += length;
948
949 before = current;
950 }
951
952 totalLength = total;
953 continuousLength = continuous;
954
955 return continuous/total;
956}

1002{
1003 auto graph = new TGraph();
1004 auto scale = 1.;
1005
1006 LKVector3 center(fA,fI,fJ,0);
1007 center = LKVector3(center.GetXYZ(), LKVector3::kZ);
1008
1009 Double_t rms = scalerms;
1010
1011 for (Double_t r = 0.; r < 1.001; r += 0.02) {
1012 auto pos = LKVector3(ExtrapolateByRatio(r),LKVector3::kZ);
1013 auto perp = LKVector3(TVector3(pos-center).Unit(), LKVector3::kZ);
1014 pos = scale * (pos + rms * perp);
1015 graph -> SetPoint(graph->GetN(), pos.At(axis1), pos.At(axis2));
1016 }
1017
1018 for (Double_t r = 1.; r >= 0; r -= 0.02) {
1019 auto pos = LKVector3(ExtrapolateByRatio(r),LKVector3::kZ);
1020 auto perp = LKVector3(TVector3(pos-center).Unit(), LKVector3::kZ);
1021 pos = scale * (pos - rms * perp);
1022 graph -> SetPoint(graph->GetN(), pos.At(axis1), pos.At(axis2));
1023 }
1024
1025 auto pos = LKVector3(ExtrapolateByRatio(0),LKVector3::kZ);
1026 auto perp = LKVector3(TVector3(pos-center).Unit(), LKVector3::kZ);
1027 pos = scale * (pos + rms * perp);
1028 graph -> SetPoint(graph->GetN(), pos.At(axis1), pos.At(axis2));
1029
1030 return graph;
