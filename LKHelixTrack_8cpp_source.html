395void LKHelixTrack::DetermineParticleCharge(TVector3 vertex)
396{
397 Double_t lHead = ExtrapolateToAlpha(fH);
398 Double_t lTail = ExtrapolateToAlpha(fT);
399
400 TVector3 q;
401 Double_t alpha;
402 Double_t lVertex = ExtrapolateToPointAlpha(vertex, q, alpha);
403
404 if (std::abs(lVertex - lTail) > std::abs(lVertex - lHead)) {
405 auto a = fT;
406 fT = fH;
407 fH = a;
408 }
409
410 if (fT < fH) {
411 if (fS > 0) fIsPositiveChargeParticle = false;
412 else fIsPositiveChargeParticle = true;
413 }
414 else {
415 if (fS > 0) fIsPositiveChargeParticle = true;
416 else fIsPositiveChargeParticle = false;
417 }

859Double_t
860LKHelixTrack::ExtrapolateByMap(TVector3 p, TVector3 &q, TVector3 &m) const
861{
862 Double_t lHead = ExtrapolateToAlpha(fH);
863 Double_t lTail = ExtrapolateToAlpha(fT);
864 Double_t lOff = lHead;
865 if (lHead > lTail)
866 lOff = lTail;
867
868 Double_t alpha;
869 Double_t length = ExtrapolateToPointAlpha(p, q, alpha);
870
871 LKVector3 p2(p, fA);
872 LKVector3 q2(q, fA);
873
874 Double_t di = p2.I() - fI;
875 Double_t dj = p2.J() - fJ;
876 Double_t dr = sqrt(di*di + dj*dj) - fR;
877 Double_t dk = p2.K() - q2.K();
878
879 m = TVector3(dr, dk/CosDip(), length+dk*TMath::Sin(DipAngle()) - lOff);
880
881 return alpha * fR / CosDip();
882}

884double LKHelixTrack::Continuity(double &totalLength, double &continuousLength, double distCut)
885{
886 Int_t numHits = fHitArray.GetNumHits();
887 if (numHits < 2)
888 return -1;
889
890 SortHits();
891
892 Double_t total = 0;
893 Double_t continuous = 0;
894 TVector3 before = Map(fHitArray.GetHit(0)->GetPosition());
895
896 for (auto iHit = 1; iHit < numHits; iHit++)
897 {
898 TVector3 current = Map(fHitArray.GetHit(iHit)->GetPosition());
899 auto length = std::abs(current.Z()-before.Z());
900
901 total += length;
902 if (length < distCut)
903 continuous += length;
904
905 before = current;
906 }
907
908 totalLength = total;
909 continuousLength = continuous;
910
911 return continuous/total;
912}

951{
952 auto graph = new TGraph();

954

962 auto perp = LKVector3(TVector3(pos-center).Unit(), LKVector3::kZ);
963 pos = scale * (pos + rms * perp);

965 }
966

968 auto pos = LKVector3(ExtrapolateByRatio(r),LKVector3::kZ);

970 pos = scale * (pos - rms * perp);
971 graph -> SetPoint(graph->GetN(), pos.At(axis1), pos.At(axis2));
972 }
973
974 auto pos = LKVector3(ExtrapolateByRatio(0),LKVector3::kZ);
975 auto perp = LKVector3(TVector3(pos-center).Unit(), LKVector3::kZ);
976 pos = scale * (pos + rms * perp);

978
