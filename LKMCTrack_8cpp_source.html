148Double_t LKMCTrack::TrackLength() const
149{
150 Int_t nPairs = fVX.size() - 1;
151 if (nPairs < 2)
152 return 0.;
153
154 Double_t length = 0.;
155
156 for (auto iPair = 0; iPair < nPairs; ++iPair) {
157 TVector3 a(fVX[iPair], fVY[iPair], fVZ[iPair]);
158 TVector3 b(fVX[iPair+1], fVY[iPair+1], fVZ[iPair+1]);
159 length += (b-a).Mag();
160 }
161
162 return length;
163}
