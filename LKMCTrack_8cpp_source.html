118Double_t LKMCTrack::TrackLength() const
119{
120 Int_t nPairs = fVX.size() - 1;
121 if (nPairs < 2)
122 return 0.;
123
124 Double_t length = 0.;
125
126 for (auto iPair = 0; iPair < nPairs; ++iPair) {
127 TVector3 a(fVX[iPair], fVY[iPair], fVZ[iPair]);
128 TVector3 b(fVX[iPair+1], fVY[iPair+1], fVZ[iPair+1]);
129 length += (b-a).Mag();
130 }
131
132 return length;
133}
