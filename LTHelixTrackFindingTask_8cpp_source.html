862{

877 auto trackHits = track -> GetHitArray();

879 track -> ExtrapolateByMap(pPre, qPre, mPre);
880 kqPre = LKVector3(qPre,fReferenceAxis);

883 auto axis2 = fPadPlane -> GetAxis2();
884

887 auto pCur = trackHits->GetHit(iHit)->GetPosition();
888 track -> ExtrapolateByMap(pCur, qCur, mCur);

891
892 auto val1 = kqCur.At(axis1) - kqPre.At(axis1);
893 auto val2 = kqCur.At(axis2) - kqPre.At(axis2);

899 continuous += length;

901

903 }
