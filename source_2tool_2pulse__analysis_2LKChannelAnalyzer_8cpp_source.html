579 for (int iTb=0; iTb<fNumTbSampleLast; iTb++) {
580 pedestalFinal += buffer[tbGlobal];
581 tbGlobal++;
582 }
583#ifdef DEBUG_CHANA_FINDPED

585#endif
586 }
587
588#ifdef DEBUG_CHANA_FINDPED
589 lk_debug << pedestalFinal << " " << countNumPedestalTb << endl;;
590#endif
591 if (pedestalFinal!=0||countNumPedestalTb!=0)

593 else
594 pedestalFinal = 4096;
595
596#ifdef DEBUG_CHANA_FINDPED

598#endif
599
600 for (auto iTb=0; iTb<fTbMax; ++iTb)

602
603 fDynamicRange = fDynamicRange - pedestalFinal;
604
605 fPedestal = pedestalFinal;
606
607 return pedestalFinal;

881void LKChannelAnalyzer::FitAmplitude(double *buffer, double tbStartOfPulse,
882 int &ndf,
883 double &amplitude,
884 double &chi2NDF)
885{
886 if (std::isnan(tbStartOfPulse)) {
887#ifdef DEBUG_CHANA_FITAMPLITUDE
888 lk_debug << "return tb is nan" << endl;
889#endif
890 chi2NDF = 1.e10;
891 return;
892 }
893#ifdef DEBUG_CHANA_FITAMPLITUDE
894 lk_debug << "pulse y-values: "; for (auto i=0; i<20; ++i) e_cout << i << "/" << fPulse -> Eval(i) << ", "; e_cout << endl;
895#endif
896 double refy = 0;
897 double ref2 = 0;
898 int tb0 = int(tbStartOfPulse);
899#ifdef DEBUG_CHANA_FITAMPLITUDE
900 lk_debug << "tb0 = abs(tbStartOfPulse) : " << tb0 << " = abs(" << tbStartOfPulse << ")" << endl;
901#endif
902 double tbOffset = tbStartOfPulse - tb0;
903
904 int ndfFit = 0;
905 int iTbPulse = 0;
906 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
907 {
908 int tbData = tb0 + iTbPulse;
909#ifdef DEBUG_CHANA_FITAMPLITUDE
910 lk_debug << "tbData = tb0 + iTbPulse : " << tbData << " = " << tb0 << " + " << iTbPulse << endl;
911#endif
912 //if (tbData<0 || tbData>=350)
913 // lk_debug << tbData << endl;
914 if (tbData>=fTbMax)
915 break;
916 double valueData = buffer[tbData];
917 if (valueData>=fDynamicRange)
918 continue;
919 ndfFit++;
920 double tbRef = iTbPulse - tbOffset + 0.5;
921 double valueRef = fPulse -> Eval(tbRef);
922 double errorRef = fPulse -> Error(tbRef);
923 double weigth = 1./(errorRef*errorRef);
924 refy += weigth * valueRef * valueData;
925 ref2 += weigth * valueRef * valueRef;
926#ifdef DEBUG_CHANA_FITAMPLITUDE
927 lk_debug << "iTbPulse; weight, valueRef, valueData : " << weigth << ", " << valueRef << ", " << valueData << endl;
928 lk_debug << "iTbPulse; amplitude = refy / ref : " << iTbPulse << "; " << refy/ref2 << " = " << refy << " / " << ref2 << endl;
929#endif
930 }
931 ndfFit = ndfFit-2;
932#ifdef DEBUG_CHANA_FITAMPLITUDE
933 lk_debug << "ndf is " << ndf << " ndfFit is " << ndfFit << endl;
934#endif
935
936 if (ref2==0) {
937 chi2NDF = 1.e10;
938 return;
939 }
940 amplitude = refy / ref2;
941#ifdef DEBUG_CHANA_FITAMPLITUDE
942 lk_debug << "final amplitude = " << amplitude << " = " << refy << " / " << ref2 << endl;
943#endif
944 chi2NDF = 0;
945
946 ndfFit = 0;
947 iTbPulse = 0;
948 for (; (ndfFit<ndf && iTbPulse<fNDFPulse); iTbPulse++)
949 {
950 int tbData = tb0 + iTbPulse;
951 //if (tbData<0 || tbData>=350)
952 // lk_debug << "e " << tbData << endl;
953 if (tbData>=fTbMax)
954 break;
955 double valueData = buffer[tbData];
956 if (valueData>=fDynamicRange)
957 continue;
958 ndfFit++;
959 double tbRef = iTbPulse - tbOffset + 0.5;
960 double valueRefA = amplitude * fPulse -> Eval(tbRef);
961 double errorRefA = amplitude * fPulse -> Error(tbRef);
962 double residual = (valueData-valueRefA)*(valueData-valueRefA)/errorRefA/errorRefA;
963#ifdef DEBUG_CHANA_FITAMPLITUDE
964 lk_debug << "residual = (valueData - valueRefA)^2 / errorRefA^2 at tbRef : " << residual << " = (" << valueData << " - " << valueRefA << ")^2 / " << errorRefA << "^2 at " << tbRef << endl;
965#endif
966 chi2NDF += residual;
967 }
968 ndfFit = ndfFit-2;
969
970 //lk_debug << "ndfFit: " << ndfFit << endl;
971
972 chi2NDF = chi2NDF/ndfFit;
973 ndf = ndfFit;
974}
