124{

134 // Fill electrons into pad plane

137 for (Long64_t iStep = 0; iStep < numMCSteps; ++iStep)
138 {
139 LKMCStep* step = (LKMCStep*) mcArray -> At(iStep);
140 Int_t trackID = step -> GetTrackID();
141 Double_t edep = step -> GetEdep();
142 TVector3 posMC(step -> GetX(), step -> GetY(), step -> GetZ());
143
144 TVector3 posFinal;
145 double driftLength;
146 padPlane -> DriftElectron(posMC, posFinal, driftLength);
147 TVector3 xyl(posFinal.X(),posFinal.Y(),driftLength);
148 auto xylDiff = sim -> CalculateGasDiffusion(driftLength);
149 auto xylGas = xyl + xylDiff;
150
151 sim -> SetNumElectrons(edep);
152 while(int numElectrons = sim -> GetNextElectronBunch())
153 {
154 if (numElectrons==0)
155 break;
156
157 auto xPlane = xylGas.X();
158 auto yPlane = xylGas.Y();
159
160 auto tb = sim -> GetTimeBucket(xylGas.Z());
161 auto xyc = sim -> CalculateGEMDiffusion(xPlane, yPlane, numElectrons);
162
163 xPlane = xPlane + xyc.X();
164 yPlane = yPlane + xyc.Y();
165 auto charge = xyc.Z();
166 padPlane -> FillPlane(xPlane, yPlane, tb, charge, trackID);
167 }
168 }
170
171
173 // Write active pad to output file
175 Int_t idx = 0;
176 LKPad *pad;
177 TIter itChannel(padPlane -> GetChannelArray());
178 while ((pad = (LKPad *) itChannel.Next())) {
179 if (pad -> IsActive() == false)
180 continue;
181 auto padSave = (LKPad*) padArray -> ConstructedAt(idx);
182 padSave -> SetPad(pad);
183 padSave -> CopyPadData(pad);
