33class LKTbIterationParameters
34{
35 public:
36 LKTbIterationParameters() {}
37
38 double fC1 = DBL_MAX;
39 double fC2 = DBL_MAX;
40 double fT1 = 0;
41 double fT2 = 0;
42 double fScaleTbStep0 = 1;
43 double fScaleTbStep = 1;
44
45 void SetScaleTbStep(double value) {
46 fScaleTbStep0 = value;
47 fScaleTbStep = value;
48 }
49
50 double Slope() const {
51 double slope = - (fC1 - fC2) / (fT1 - fT2);
52 return slope;
53 }
54
55 double TbStep() const {
56 double tbStep = fScaleTbStep * Slope();
57 return tbStep;
58 }
59
60 double NextTb(double tb) const {
61 double step = TbStep();
62 if (step> 1) step = 1;
63 else if (step<-1) step = -1;
64 double tbNext = tb + step;
65 return tbNext;
66 }
67
68 bool Add(double c0, double t0) {
69 if (c0<fC1) {
70 fC2 = fC1;
71 fT2 = fT1;
72 fC1 = c0;
73 fT1 = t0;
74 fScaleTbStep = fScaleTbStep0;
75 return true;
76 }
77 else if (c0<fC2) {
78 fC2 = c0;
79 fT2 = t0;
80 fScaleTbStep = fScaleTbStep0;
81 return true;
82 }
83 else {
84 fScaleTbStep = 0.5 * fScaleTbStep;
85 return false;
86 }
87 }
