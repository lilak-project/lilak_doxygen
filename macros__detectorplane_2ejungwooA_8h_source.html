212TCanvas *LKWindowManager::CanvasR(const char *name, Int_t w, Int_t h, Int_t ww, Int_t wh, Double_t rw, Double_t rh)
213{
214 if (w<0) { w = fWCurrent; }
215 if (h<0) { h = fHCurrent; }
216
217 if (ww<=0 && wh<=0) {
218 ww = fDWCanvas;
219 wh = fDHCanvas;
220 }
221 else if (ww<=0) { ww = wh * (double(fDWCanvas)/fDHCanvas); }
222 else if (wh<=0) { wh = ww * (double(fDHCanvas)/fDWCanvas); }
223
224 Int_t dwFinal = fDWCanvas;
225 Int_t dhFinal = fDHCanvas;
226
227 if (rw>1) rw = 0.01*rw;
228 if (rh>1) rh = 0.01*rh;
229 if (rw>0 && rw<=1 && rh>0 && rh<=1) {
230 dwFinal = rw * fDWFull;
231 dhFinal = rh * fDHFull;
232 }
233 else if (rw>0 && rw<=1) {
234 dwFinal = rw * fDWFull;
235 dhFinal = dwFinal * (double(wh)/ww);
236 }
237 else if (rh>0 && rh<=1) {
238 dhFinal = rh * fDHFull;
239 dwFinal = dhFinal * (double(ww)/wh);
240 }
241
242 dwFinal = dwFinal * fWGlobalScale;
243 dhFinal = dhFinal * fHGlobalScale;
244
245 //cout << dwFinal << " " << dhFinal << endl;
246 auto cvs = new TCanvas(name, name, w, h, dwFinal, dhFinal);
247
248 fWCurrent = fWCurrent + fDWSpacing;
249 fHCurrent = fHCurrent + fDHSpacing;
250
251 return cvs;
252}

613TH2D *ejungwoo::Frame(TGraph *graph, const char* name, const char* title="", ejungwoo::Range rx1, ejungwoo::Range rx2, ejungwoo::Range ry1, ejungwoo::Range ry2)
614{
615 int nbins = graph -> GetN();
616 double x1=DBL_MAX, x2=-DBL_MAX, y1=DBL_MAX, y2=-DBL_MAX;
617 for (auto iPoint=0; iPoint<nbins; ++iPoint) {
618 double x0, y0;
619 graph -> GetPoint(iPoint,x0,y0);
620 if (x0<x1) x1 = x0;
621 if (x0>x2) x2 = x0;
622 if (y0<y1) y1 = y0;
623 if (y0>y2) y2 = y0;
624 }
625 x1 = x1 - (x2-x1) * 0.05;
626 x2 = x2 + (x2-x1) * 0.05;
627 y1 = y1 - (y2-y1) * 0.05;
628 y2 = y2 + (y2-y1) * 0.05;
629 //lk_debug << "x: " << x1 << " " << x2 << ", y:" << y1 << " " << y2 << endl;
630 x1 = rx1.SetValue(x1);
631 y1 = ry1.SetValue(y1);
632 x2 = rx2.SetValue(x2);
633 y2 = ry2.SetValue(y2);
634 //lk_debug << "x: " << x1 << " " << x2 << ", y:" << y1 << " " << y2 << endl;
635 auto hist = new TH2D(name,title,100,x1,x2,200,y1,y2);
636 return hist;
