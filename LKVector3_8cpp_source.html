53LKVector3::Axis LKVector3::GetGlobalAxis(Axis axisIn) const
54{
55 if (IsGlobalAxis(axisIn))
56 return axisIn;
57
58 Axis axisOut = kNon;
59 Axis axisRef = fReferenceAxis;
60 if (axisRef==kNon)
61 return kNon;
62
63 bool negativeCorrection = 0;
64 if (int(axisIn)%2==0) {
65 negativeCorrection = 1;
66 axisIn = LKVector3::Axis(int(axisIn)-1);
67 }
68 if (int(axisRef)%2==0) {
69 negativeCorrection = !negativeCorrection;
70 axisRef = LKVector3::Axis(int(axisRef)-1);
71 }
72
73 if (axisRef==kZ) { axisOut = ( (axisIn==kI) ?kX :((axisIn==kJ) ?kY :kZ) ); }
74 else if (axisRef==kY) { axisOut = ( (axisIn==kI) ?kZ :((axisIn==kJ) ?kX :kY) ); }
75 else if (axisRef==kX) { axisOut = ( (axisIn==kI) ?kY :((axisIn==kJ) ?kZ :kX) ); }
76 else {}
77
78 axisOut = Axis(int(axisOut)+int(negativeCorrection));
79 return axisOut;
80}
