194int GetType(int cobo, int asad, int aget, int chan)
195{
196 if (cobo==0 && (asad==0 || asad==1)) return eMMCenter;
197 if (cobo==0 && asad==2 && (aget==0 || aget==1)) return eMMLeftSide;
198 if (cobo==0 && asad==2 && (aget==2 || aget==3)) return eMMLeftCenter;
199 if (cobo==0 && asad==3 && (aget==0 || aget==1)) return eMMRightSide;
200 if (cobo==0 && asad==3 && (aget==2 || aget==3)) return eMMRightCenter;
201 if (cobo==1 && asad==0 && aget==0) return efSiJunction;
202 if (cobo==1 && asad==0 && aget==1) return efSiOhmic;
203 if (cobo==1 && asad==1) return efCsI;
204 if (cobo==2 && aget==0) return eX6Ohmic;
205 if (cobo==2 && (aget==1 || aget==2)) return eX6Junction;
206 if (cobo==2 && aget==3) return eCsICT;
207
208 return -1;
209}
