111LKPulseGenerator::Pulse(Double_t x, Double_t amp, Double_t tb0)
112{
113 if (fIsDeltaFunction)
114 {
115 if (x>=tb0-1 && x<tb0) {
116 auto val = amp * (x - tb0 + 1);
117 //return val;

120 else if (x>=tb0 && x<tb0+1) {
121 auto val = - amp * (x - tb0 - 1);
122 //return val;
123 return amp;

125 else
126 return 0;

128
129 Double_t tb = x - tb0;
130 if (tb < 0)
131 return 0;
132
133 Int_t tbInStep = tb / fStepSize;
134 if (tbInStep > fNumDataPoints - 2)
135 return 0;
136
137 Double_t r = (tb / fStepSize) - tbInStep;
138 Double_t val = r * fPulseData[tbInStep + 1].fValue + (1 - r) * fPulseData[tbInStep].fValue;
139
140 return amp * val;
