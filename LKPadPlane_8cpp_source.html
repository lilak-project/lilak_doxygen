364{
365 vector<LKPad *> neighborsUsed;

370 if (id < 0)

379 neighborsNew.clear();
380 GrabNeighborPads(&neighborsTemp, &neighborsNew);

468{
469 lk_info << "Number of pads: " << fChannelArray -> GetEntries() << endl;
470
471 Int_t countM1 = 0;
472 Int_t countBad = 0;
473
474 LKPad *pad;
475 TIter iterPads(fChannelArray);

477 if (pad -> GetPadID() == -1) {

480 }

482 auto padID0 = pad -> GetPadID();
483 auto padID1 = FindPadID(center0.I(),center0.J());

489 << " --> Pad:" << padID1 << "(" << center1.I() << "," << center1.J() << "|" << pad1-> GetSection() << "," << pad1-> GetRow() << "," << pad1-> GetLayer() << ")" << endl;
490 ++countBad;
491 }
492 if (checkCorners) {
493 for (auto corner : *(pad->GetPadCorners())) {
494 auto pos = 0.1*TVector2(center0.I(),center0.J()) + 0.9*corner;
495 padID1 = FindPadID(pos.X(),pos.Y());
496 if (padID1 != padID0) {
497 auto pad1 = (LKPad *) fChannelArray -> At(padID1);
498 auto center1 = pad1 -> GetPosition();
499 lk_info << " Corner(" << pos.X() << "," << pos.Y() << ")"
500 << " --> Pad:" << padID1 << "(" << center1.I() << "," << center1.J() << "|" << pad1-> GetSection() << "," << pad1-> GetRow() << "," << pad1-> GetLayer() << ")" << endl;
501 ++countBad;
502 }
503 }
504 }
505 }
506
507 lk_info << "=================== Number of 'id = -1' pads: " << countM1 << endl;
