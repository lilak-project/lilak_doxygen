208 continue;

210 double value = (fChannelData[tb] - fPedestal) * scale;
211 fHistPedestalResidual -> Fill(value);
212 fAverageData[tb_aligned] += value;

226
228
229 fCountGoodBackgrounds++;
230 for (auto tb : tbBackground) {
231 fCountBGBin[tb] += 1;
232 fBackground[tb] += (fChannelData[tb] - fPedestal);
233 }
234
236 }
237 }
238
239 fHistPedestalPry -> Fill(fPedestalPry);
240
241 fTree -> Fill();

890double LKPulseAnalyzer::FullWidthRatioMaximum(TH1D *hist, double ratioFromMax, double numSplitBin, double &tbLeading, double &tbTrailing, double &error)
891{
892 int numBins = hist -> GetNbinsX();
893 int binMax = hist -> GetMaximumBin();
894 double xAtMax = hist -> GetBinCenter(binMax);
895 double yMax = hist -> GetBinContent(binMax);
896 double yAtRatio = ratioFromMax * yMax;
897 double xMin = hist -> GetXaxis() -> GetBinLowEdge(1);
898 double xMax = hist -> GetXaxis() -> GetBinUpEdge(numBins);
899 double dx = (xMax-xMin)/numBins/numSplitBin;
900 double yError1 = DBL_MAX;
901 double yError0 = DBL_MAX;
902
903 double x;
904 double xLow;
905 double xHigh;
906 double yValue;
907 double yError;
908
909 for (x=xAtMax; x>=xMin; x-=dx)
910 {
911 yValue = hist -> Interpolate(x);
912 yError = TMath::Abs(yValue - yAtRatio);
913 if (yError > yError0) { xLow = x + dx; break; }
914 yError0 = yError;
915 }
916 if (x==xMin)
917 return -1;
918
919 for (x=xAtMax; x<=xMax; x+=dx)
920 {
921 yValue = hist -> Interpolate(x);
922 yError = TMath::Abs(yValue - yAtRatio);
923 if (yError > yError1) { xHigh = x - dx; break; }
924 yError1 = yError;
925 }
926 if (x==xMax)
927 return -1;
928
929 tbLeading = xLow;
930 tbTrailing = xHigh;
931 error = TMath::Sqrt(yError0*yError0 + yError1*yError1);
932 double width = xHigh - xLow;
933
934 return width;
935}
