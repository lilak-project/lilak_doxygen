212TCanvas *LKWindowManager::CanvasR(const char *name, Int_t w, Int_t h, Int_t ww, Int_t wh, Double_t rw, Double_t rh)
213{
214 if (w<0) { w = fWCurrent; }
215 if (h<0) { h = fHCurrent; }
216
217 if (ww<=0 && wh<=0) {
218 ww = fDWCanvas;
219 wh = fDHCanvas;
220 }
221 else if (ww<=0) { ww = wh * (double(fDWCanvas)/fDHCanvas); }
222 else if (wh<=0) { wh = ww * (double(fDHCanvas)/fDWCanvas); }
223
224 Int_t dwFinal = fDWCanvas;
225 Int_t dhFinal = fDHCanvas;
226
227 if (rw>1) rw = 0.01*rw;
228 if (rh>1) rh = 0.01*rh;
229 if (rw>0 && rw<=1 && rh>0 && rh<=1) {
230 dwFinal = rw * fDWFull;
231 dhFinal = rh * fDHFull;
232 }
233 else if (rw>0 && rw<=1) {
234 dwFinal = rw * fDWFull;
235 dhFinal = dwFinal * (double(wh)/ww);
236 }
237 else if (rh>0 && rh<=1) {
238 dhFinal = rh * fDHFull;
239 dwFinal = dhFinal * (double(ww)/wh);
240 }
241
242 dwFinal = dwFinal * fWGlobalScale;
243 dhFinal = dhFinal * fHGlobalScale;
244
245 //cout << dwFinal << " " << dhFinal << endl;
246 auto cvs = new TCanvas(name, name, w, h, dwFinal, dhFinal);
247
248 fWCurrent = fWCurrent + fDWSpacing;
249 fHCurrent = fHCurrent + fDHSpacing;
250
251 return cvs;
252}

427TH2D *ejungwoo::Frame(TGraph *graph, const char* name, const char* title="")
428{
429 int nbins = graph -> GetN();
430 double x1=DBL_MAX, x2=-DBL_MAX, y1=DBL_MAX, y2=-DBL_MAX;
431 for (auto iPoint=0; iPoint<nbins; ++iPoint) {
432 double x0, y0;
433 graph -> GetPoint(iPoint,x0,y0);
434 if (x0<x1) x1 = x0;
435 if (x0>x2) x2 = x0;
436 if (y0<y1) y1 = y0;
437 if (y0>y2) y2 = y0;
438 }
439 x1 = x1 - (x2-x1) * 0.05;
440 x2 = x2 + (x2-x1) * 0.05;
441 y1 = y1 - (y2-y1) * 0.05;
442 y2 = y2 + (y2-y1) * 0.05;
443 auto hist = new TH2D(name,title,100,x1,x2,200,y1,y2);
444 return hist;
445}

447TF1 *ejungwoo::FitG(TH1D* hist, double sigmaWidth=1.5, Option_t *opt="RQ0")
448{
449 auto binMax = hist -> GetMaximumBin();
450 auto max = hist -> GetBinContent(binMax);
451 auto xMax = hist -> GetXaxis() -> GetBinCenter(binMax);
452 auto xErr = hist -> GetStdDev();
453 auto func = new TF1(Form("%s_fitg",hist->GetName()),"gaus(0)",xMax-xErr*sigmaWidth,xMax+xErr*sigmaWidth);
454 func -> SetParameters(max,xMax,xErr);
455 hist -> Fit(func,opt);
456 xMax = func -> GetParameter(1);
457 xErr = func -> GetParameter(2);
458 func -> SetRange(xMax-sigmaWidth*xErr,xMax+sigmaWidth*xErr);
459 hist -> Fit(func,opt);
460 return func;
461}
