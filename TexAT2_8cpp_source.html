472int TexAT2::GetElectronicsID(int cobo, int asad, int aget, int chan)
473{
474 if(chan==11 || chan==22 || chan==45 || chan==56) return -1;
475
476 if (cobo==0 && (asad==0 || asad==1)) {
477 int mchannel = chan;
478 if(chan<11) mchannel = chan;
479 else if(chan<22) mchannel = chan - 1;
480 else if(chan<45) mchannel = chan - 2;
481 else if(chan<56) mchannel = chan - 3;
482 else mchannel = chan - 4;
483
484 if ((int(mchannel+1)/2)%2==0)
485 {
486 if (aget==0) return eMMCenterSideA0;
487 else if (aget==1) return eMMCenterSideA1;
488 else if (aget==2) return eMMCenterSideA2;
489 else if (aget==3) return eMMCenterSideA3;
490 }
491 else
492 {
493 if (aget==0) return eMMCenterCenterA0;
494 else if (aget==1) return eMMCenterCenterA1;
495 else if (aget==2) return eMMCenterCenterA2;
496 else if (aget==3) return eMMCenterCenterA3;
497 }
498 }
499 if (cobo==0 && asad==2 && (aget==0 || aget==1)) return eMMLeftSide;
500 if (cobo==0 && asad==2 && (aget==2 || aget==3)) return eMMLeftCenter;
501 if (cobo==0 && asad==3 && (aget==0 || aget==1)) return eMMRightSide;
502 if (cobo==0 && asad==3 && (aget==2 || aget==3)) return eMMRightCenter;
503 if (cobo==1 && asad==0 && aget==0) return efSiJunction;
504 if (cobo==1 && asad==0 && aget==1) return efSiOhmic;
505 if (cobo==1 && asad==1 && aget==0
506 && (chan==2
507 || chan==7
508 || chan==10
509 || chan==16
510 || chan==19
511 || chan==25
512 || chan==28
513 || chan==33
514 || chan==36
515 || chan==41)
516 )
517 return efCsI;
518 if (cobo==2 && aget==0) return eX6Ohmic;
519 if (cobo==2 && (aget==1 || aget==2)) return eX6Junction;
520 if (cobo==2 && asad==1 && aget==3) return eCsICT;
521
522 return -1;
523}
