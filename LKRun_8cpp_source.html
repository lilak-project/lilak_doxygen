706{
707 if (fBranchPtrMap[name] != nullptr)

721 else
722 persistencyMessage = "(temporary)";
723 }

725 fBranchPtr[fCountBranches] = obj;
726 fBranchPtrMap[name] = fBranchPtr[fCountBranches];
727 fBranchNames.push_back(name);
728 fCountBranches++;
729
730 if (persistent) {
731 if (fOutputTree != nullptr)
732 fOutputTree -> Branch(name, obj, 32000, 0);
733 fPersistentBranchArray -> Add(obj);
734 } else {
735 fTemporaryBranchArray -> Add(obj);
736 }
737 lk_info << "Output branch " << name << " " << persistencyMessage << endl;

944bool LKRun::StartOfRun(Long64_t numEvents)
945{
946 if (fInitialized == false) {
947 lk_info << "LKRun is not initialized!" << endl;
948 lk_info << "try initialization..." << endl;
949 if (!Init()) {
950 lk_error << "Exit Run() due to initialization fail." << endl;
951 return false;
952 }
953 }
954
955 if (fNumEntries<=0)
956 fNumEntries = numEvents;
957
958 CheckIn();
959
960 if (numEvents > 0)
961 fEndEventID = numEvents - 1;
962
963 if (fStartEventID == -1)
964 fStartEventID = 0;
965
966 if (fEndEventID == -1)
967 fEndEventID = fNumEntries-1;
968
969 if (fSplit >= 0) {
970 fStartEventID = fSplit * fNumSplitEntries;
971 fEndEventID = ((fSplit+1) * fNumSplitEntries) - 1 ;
972 if (fEndEventID > fNumEntries - 1)
973 fEndEventID = fNumEntries - 1;
974 }
975
976 fNumRunEntries = fEndEventID - fStartEventID + 1;
977
978 if (fNumRunEntries<fNumPrintMessage)
979 fNumSkipEventsForMessage = 1;
980 fNumSkipEventsForMessage = fNumRunEntries/fNumPrintMessage;
981 if (fNumSkipEventsForMessage==0)
982 fNumSkipEventsForMessage = 1;
983
984 return true;
985}
