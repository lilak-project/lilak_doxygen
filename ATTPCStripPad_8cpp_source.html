390{
391 std::vector<Int_t> indexArray;
392
393 for(int i=0; i<3; i++){
394 Double_t thisPadX = posPad.I();
395 Double_t thisPadY = posPad.J();
396
397 int xIndex = i%3 -1;
398 int yIndex = i%2 +1;
399 if(i>=3){yIndex= -yIndex;}
400 Double_t neighborPadX = thisPadX + xIndex*(sqrt(3)*fPadGap/2. +3.*fPadWidth/2.);
401 Double_t neighborPadY = thisPadY + yIndex*(fPadGap/2. + fPadHeight/2.);
402
403 Int_t neighborIndex = PadIndexCheck(neighborPadX, neighborPadY);
404 if(neighborIndex > 0){
405 indexArray.push_back(neighborIndex);
406 }
