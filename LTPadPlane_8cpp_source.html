461int LTPadPlane::FindPadID(double i, double j)
462{
463 int section = FindSection(i,j);
464
465 double xRotatedToSec0 = i*fCosPiNo4[section] + j*fSinPiNo4[section];
466 double yRotatedToSec0 = -i*fSinPiNo4[section] + j*fCosPiNo4[section];
467 double rRotatedToSec0 = sqrt(xRotatedToSec0*xRotatedToSec0 + yRotatedToSec0*yRotatedToSec0);
468
469 double rFromSectionBottom = rRotatedToSec0 - fRadiusLayer0 + .5*fPadHeight;
470 if (rFromSectionBottom < 0)
471 return -1;
472
473 int layer = (int)(rFromSectionBottom/fRSpacing);
474 if (layer > fNumLayers)
475 return -2;
476
477 int pm = 1;
478 if (xRotatedToSec0 < 0) {
479 xRotatedToSec0 = -xRotatedToSec0;
480 pm = -1;
481 }
482
483 if (xRotatedToSec0 < .5*fPadGap)
484 return -3;
485
486 double xFromRow0LeftEdge = xRotatedToSec0 - .5*fPadGap;
487 int row = (int) (xFromRow0LeftEdge / fXSpacing) + 1;
488 if (xFromRow0LeftEdge - (row-1)*fXSpacing > fPadWidth)
489 return -4;
490
491 row = row - fNumSkippedHalfRows[layer];
492
493 if (row > fNumHalfRowsInLayer[layer])
494 return -5;
495
496 return FindPadID(section,layer,pm*row);
497}
