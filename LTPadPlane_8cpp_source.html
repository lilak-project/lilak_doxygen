390Int_t LTPadPlane::FindPadID(Double_t i, Double_t j)
391{
392 Int_t section = FindSection(i,j);
393
394 Double_t xRotatedToSec0 = i*fCosPiNo4[section] + j*fSinPiNo4[section];
395 Double_t yRotatedToSec0 = -i*fSinPiNo4[section] + j*fCosPiNo4[section];
396 Double_t rRotatedToSec0 = sqrt(xRotatedToSec0*xRotatedToSec0 + yRotatedToSec0*yRotatedToSec0);
397
398 Double_t rFromSectionBottom = rRotatedToSec0 - fRadiusLayer0 + .5*fPadHeight;
399 if (rFromSectionBottom < 0)
400 return -1;
401
402 Int_t layer = (Int_t)(rFromSectionBottom/fRSpacing);
403 if (layer > fNumLayers)
404 return -2;
405
406 Int_t pm = 1;
407 if (xRotatedToSec0 < 0) {
408 xRotatedToSec0 = -xRotatedToSec0;
409 pm = -1;
410 }
411
412 if (xRotatedToSec0 < .5*fPadGap)
413 return -3;
414
415 Double_t xFromRow0LeftEdge = xRotatedToSec0 - .5*fPadGap;
416 Int_t row = (Int_t) (xFromRow0LeftEdge / fXSpacing) + 1;
417 if (xFromRow0LeftEdge - (row-1)*fXSpacing > fPadWidth)
418 return -4;
419
420 row = row - fNumSkippedHalfRows[layer];
421
422 if (row > fNumHalfRowsInLayer[layer])
423 return -5;
424
425 return FindPadID(section,layer,pm*row);
