329double LKChannelAnalyzer::FullWidthRatioMaximum(TH1D *hist, double ratioFromMax, double numSplitBin, double &x0, double &x1, double &error)
330{
331 int numBins = hist -> GetNbinsX();
332 int binMax = hist -> GetMaximumBin();
333 double xAtMax = hist -> GetBinCenter(binMax);
334 double yMax = hist -> GetBinContent(binMax);
335 double yAtRatio = ratioFromMax * yMax;
336 double xMin = hist -> GetXaxis() -> GetBinLowEdge(1);
337 double xMax = hist -> GetXaxis() -> GetBinUpEdge(numBins);
338 double dx = (xMax-xMin)/numBins/numSplitBin;
339 double yError1 = DBL_MAX;
340 double yError0 = DBL_MAX;
341
342 double x;
343 double xLow;
344 double xHigh;
345 double yValue;
346 double yError;
347
348 for (x=xAtMax; x>=xMin; x-=dx)
349 {
350 yValue = hist -> Interpolate(x);
351 yError = TMath::Abs(yValue - yAtRatio);
352 if (yError > yError0) { xLow = x + dx; break; }
353 yError0 = yError;
354 }
355 if (x==xMin)
356 return -1;
357
358 for (x=xAtMax; x<=xMax; x+=dx)
359 {
360 yValue = hist -> Interpolate(x);
361 yError = TMath::Abs(yValue - yAtRatio);
362 if (yError > yError1) { xHigh = x - dx; break; }
363 yError1 = yError;
364 }
365 if (x==xMax)
366 return -1;
367
368 x0 = xLow;
369 x1 = xHigh;
370 error = TMath::Sqrt(yError0*yError0 + yError1*yError1);
371 double width = xHigh - xLow;
372
373 return width;
